## 1. 適配器框架概述

### 1.1 設計目標

xCloudVLMUI 的適配器框架（Adapter Framework）旨在解決以下問題：

1. **硬體多樣性**：工廠現場感測器品牌繁多，協議各異（MQTT、Modbus、OPC-UA、REST）
2. **LLM 替換性**：需能在不更改業務邏輯的情況下切換 LLM 後端（llama.cpp、Ollama、OpenAI）
3. **儲存擴展性**：支援未來從 SQLite 遷移至 PostgreSQL 或 TimescaleDB

### 1.2 適配器模式架構

```
┌─────────────────────────────────────────────────────────────┐
│                      業務層（Services）                     │
│   VHSService | PipelineService | AlertService | ...         │
└──────────────────────────┬──────────────────────────────────┘
                           │ 呼叫抽象介面
┌──────────────────────────▼──────────────────────────────────┐
│                    適配器介面（Protocols）                   │
│   ISensorAdapter | ILLMAdapter | IStorageAdapter | ...      │
└──────────────────────────┬──────────────────────────────────┘
                           │ 依賴注入（FastAPI Depends）
┌──────────────────────────▼──────────────────────────────────┐
│                    具體適配器實作                            │
│                                                             │
│  感測器：                                                    │
│   MQTTAdapter（已實作）| ModbusAdapter（v1.2）              │

└─────────────────────────────────────────────────────────────┘
```

```python
    """根據 settings.LLM_BACKEND 回傳對應適配器"""
```

### 3.1 MQTTAdapter（v1.1.0）

```python
class MQTTAdapter:
    """
    Eclipse Mosquitto MQTT 適配器
    
    Topic 格式: xcloud/{device_id}/{sensor_type}
    QoS: 1（at least once）
    TLS: 可選（設定於 MQTTConfig）
    """
    
    PROTOCOL = "MQTT 3.1.1"
    DEFAULT_PORT = 1883
    DEFAULT_TLS_PORT = 8883
    
    def __init__(self, host: str, port: int = 1883):
        self._host = host
        self._port = port
        self._client = mqtt.Client(callback_api_version=CallbackAPIVersion.VERSION2)
        self._connected = False
        self._message_queue: asyncio.Queue[SensorReading] = asyncio.Queue()
    
    async def connect(self, config: dict) -> bool:
        self._client.on_connect = self._on_connect
        self._client.on_message = self._on_message
        self._client.on_disconnect = self._on_disconnect
        
        if config.get("tls_enabled"):
            self._client.tls_set(ca_certs=config.get("ca_cert_path"))
        
        if config.get("username"):
            self._client.username_pw_set(config["username"], config["password"])
        
        self._client.connect_async(self._host, self._port)
        self._client.loop_start()
        
        # 等待連線建立（最多 10 秒）
        for _ in range(100):
            if self._connected:
                return True
            await asyncio.sleep(0.1)
        return False
```

### 3.2 LlamaCppAdapter（v1.1.0）

```python
class LlamaCppAdapter:
    """
    llama.cpp HTTP Server 適配器
    
    相容 OpenAI API 格式
    支援：/v1/chat/completions（含串流）
    端點：http://llama-cpp:8080
    """
    
    def __init__(self, host: str):
        self._host = host.rstrip("/")
        self._client = httpx.AsyncClient(timeout=60.0)
    
    async def chat(
        self,
        messages: list[dict],
        config: LLMConfig | None = None
    ) -> LLMResponse:
        cfg = config or LLMConfig(model=settings.LLAMA_CPP_MODEL)
        
        start = time.time()
        resp = await self._client.post(
            f"{self._host}/v1/chat/completions",
            json={
                "model": cfg.model,
                "messages": messages,
                "temperature": cfg.temperature,
                "max_tokens": cfg.max_tokens,
                "stream": False,
            }
        )
        resp.raise_for_status()
        data = resp.json()
        
        return LLMResponse(
            content=data["choices"][0]["message"]["content"],
            model=data["model"],
            tokens_used=data["usage"]["total_tokens"],
            duration_ms=(time.time() - start) * 1000,
            finish_reason=data["choices"][0]["finish_reason"],
        )
    
    async def analyze_image(
        self,
        image_data: bytes,
        prompt: str,
        config: LLMConfig | None = None
    ) -> LLMResponse:
        """Base64 編碼圖片，送至 llama.cpp multimodal endpoint"""
        import base64
        b64_image = base64.b64encode(image_data).decode()
        
        messages = [{
            "role": "user",
            "content": [
                {"type": "image_url", "image_url": {"url": f"data:image/jpeg;base64,{b64_image}"}},
                {"type": "text", "text": prompt}
            ]
        }]
        return await self.chat(messages, config)
```

### 3.3 ChromaDBAdapter（v1.1.0）

```python
class ChromaDBAdapter:
    """
    ChromaDB 向量資料庫適配器
    
    支援：語意搜尋、文件管理
    持久化：volume mount 至 /chroma/data
    """
    
    def __init__(self, host: str, collection: str = "knowledge"):
        import chromadb
        self._client = chromadb.HttpClient(host=host)
        self._collection_name = collection
    
    @property
    def collection(self):
        return self._client.get_or_create_collection(
            name=self._collection_name,
            metadata={"hnsw:space": "cosine"}
        )
    
    async def query(
        self,
        query_text: str,
        top_k: int = 5,
        filter: dict | None = None
    ) -> list[SearchResult]:
        results = self.collection.query(
            query_texts=[query_text],
            n_results=top_k,
            where=filter,
        )
        
        return [
            SearchResult(
                doc_id=doc_id,
                content=doc,
                score=1 - distance,  # 轉換距離為相似度
                metadata=meta
            )
            for doc_id, doc, distance, meta in zip(
                results["ids"][0],
                results["documents"][0],
                results["distances"][0],
                results["metadatas"][0],
            )
        ]
```

## 4. 感測器適配器設計

### 4.1 MQTT Topic 解析規則

```python
# Topic 格式：xcloud/{device_id}/{sensor_type}
# 範例：
#   xcloud/pump001/temperature     → device_id=pump001, sensor_type=temperature
#   xcloud/motor-A3/vibration-x   → device_id=motor-A3, sensor_type=vibration-x
#   xcloud/+/temperature           → 訂閱所有設備的溫度

def parse_topic(topic: str) -> tuple[str, str]:
    parts = topic.split("/")
    if len(parts) != 3 or parts[0] != "xcloud":
        raise ValueError(f"Invalid topic format: {topic}")
    return parts[1], parts[2]  # device_id, sensor_type
```

### 4.2 Payload 標準格式

```json
{
    "value": 72.5,
    "unit": "°C",
    "timestamp": "2026-04-10T14:30:00Z",
    "quality": "good",
    "metadata": {
        "sensor_id": "TMP-001",
        "firmware": "v2.1.0"
    }
}
```

若感測器設備僅能發送簡單數值（如 `72.5`），適配器自動包裝為標準格式。

### 4.3 閾值比較邏輯

```python
def exceeds_threshold(value: float, config: MQTTConfig) -> bool:
    """
    threshold_type:
      gt  → value > threshold_value (超過上限)
      lt  → value < threshold_value (低於下限)
      eq  → value == threshold_value (等於警戒值)
      gte → value >= threshold_value
      lte → value <= threshold_value
    """
    ops = {
        "gt": lambda v, t: v > t,
        "lt": lambda v, t: v < t,
        "eq": lambda v, t: abs(v - t) < 0.001,
        "gte": lambda v, t: v >= t,
        "lte": lambda v, t: v <= t,
    }
    return ops.get(config.threshold_type, lambda v, t: False)(value, config.threshold_value)
```

### 5.2 Context Window 管理

```python
class ContextWindowManager:
    """
    管理 LLM 對話 Context Window（128K tokens for Gemma 4 E4B）
    - 超過限制時自動截斷舊訊息
    - 保留 system prompt + 最近 N 輪對話
    """
    
    MAX_TOKENS = 128_000
    RESERVE_TOKENS = 2_048    # 保留給回應
    
    def trim_messages(self, messages: list[dict], model_tokens: int) -> list[dict]:
        available = self.MAX_TOKENS - self.RESERVE_TOKENS
        system_msgs = [m for m in messages if m["role"] == "system"]
        user_msgs = [m for m in messages if m["role"] != "system"]
        
        # 從最舊的 user/assistant 訊息開始刪除
        while self._estimate_tokens(system_msgs + user_msgs) > available:
            if len(user_msgs) <= 2:  # 保留至少最後一輪
                break
            user_msgs.pop(0)
        
        return system_msgs + user_msgs
```

```python
class IEquipmentRepository(Protocol):
    async def get_all(self) -> list[Equipment]: ...
    async def get_by_id(self, id: int) -> Equipment | None: ...
    async def create(self, data: EquipmentCreate) -> Equipment: ...
    async def update(self, id: int, data: EquipmentUpdate) -> Equipment: ...
    async def delete(self, id: int) -> bool: ...

class SQLiteEquipmentRepository:
    """SQLAlchemy + SQLite 實作"""
    
    def __init__(self, session: AsyncSession):
        self._session = session
    
    async def get_all(self) -> list[Equipment]:
        result = await self._session.execute(select(EquipmentModel))
        return [row.to_domain() for row in result.scalars().all()]
```

```python
# backend/adapters/modbus_adapter.py
from pymodbus.client import AsyncModbusTcpClient
from .base import ISensorAdapter, SensorReading

class ModbusAdapter:
    """Modbus TCP 感測器適配器"""
    
    PROTOCOL = "Modbus TCP"
    
    def __init__(self, host: str, port: int = 502):
        self._client = AsyncModbusTcpClient(host, port=port)
    
    async def connect(self, config: dict) -> bool:
        return await self._client.connect()
    
    async def read_register(
        self,
        device_id: str,
        register_address: int,
        sensor_type: str,
        scale: float = 1.0,
        unit: str = ""
    ) -> SensorReading:
        result = await self._client.read_holding_registers(
            address=register_address,
            count=1,
            slave=int(device_id)
        )
        raw_value = result.registers[0] * scale
        return SensorReading(
            device_id=device_id,
            sensor_type=sensor_type,
            value=raw_value,
            unit=unit,
            timestamp=datetime.utcnow()
        )
```
